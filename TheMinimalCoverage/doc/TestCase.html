def print_mc
  if @minimal_segment_count == segment_count+1
    print "0\n";
  else
    print "#{@minimal_segment_count}\n"
    for i in 0..segment_count-1
      if is_in_mc?(i)
        get_segment(i).to_s2
      end
    end
  end
end
